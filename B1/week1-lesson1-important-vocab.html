ནམ་མཁའི་རྒྱུ་སྐར་གྱི་བགྲོད་པ་དང་སྐར་ཚོགས་ཀྱི་གོ་ས་སོགས་མཚོན་པའི་དཔེ་རིས་ལ་ནམ་མཁའི་གོ་ལ་ཟེར། ༼ས་འགུལ༽ ༼༡༽ སའི་གོ་ལའི་ནང་གི་འགྱུར་ལྡོག་གི་དབང་གིས་བྱུང་བའི་ས་ཤུན་ཀྱི་གཡོ་འགུལ་གྱི་མིང་སྟེ། ས་གཡོ་བ་ཡང་ཟེར། ༼༢༽ ཕྱིའི་རྐྱེན་གྱིས་ས་ངོས་གཡོ་འགུལ་བྱེད་པའི་དོན་ཏེ། མེ་སྒྱོགས་བརྒྱབ་ནས་ས་འགུལ་ལྟ་བུ། ༼ས་བཅད༽ དཔེ་ཆ་དང་རྩོམ་གཞུང་དོན་ཚན་གྱི་མིང་སྟེ། ལེའུ་དང་པོའི་ས་བཅད་གསུམ་པ་ཞེས་པ་ལྟ་བུ། ༼ས་ཆ༽ ༼༡༽ ཡུལ་གྲུ་དང་མངའ་ཁོངས་ཀྱི་མིང་སྟེ། ཁྱོད་ས་ཆ་གང་ནས་འོངས། ཕྱི་གླིང་གི་ས་ཆ་ཞེས་པ་ལྟ་བུ། ༼༢༽ རྩིག་ལས་སོགས་ཀྱི་གནས་ངེས་ཅན་གྱི་མིང་སྟེ། ཁང་བ་ལས་སའི་ས་ཆ་ཞེས་པ་ལྟ་བུ། ༼ས་རྒྱུས༽ འཛམ་གླིང་ཡོངས་སམ་ཡུལ་གྲུ་གཅིག་གི་རི་ལུང་དང་ནམ་ཟླ། དངོས་རྫས། བགྲོད་ལམ། གནས་མི་བཅས་ཀྱི་སྤྱི་འདུས་ཀྱི་མིང་། ༼ས་སྟེང༽ ༼༡༽ མི་དང་དུད་འགྲོ་སོགས་གནས་པའི་ཡུལ་གྱི་མིང་སྟེ། ས་སྟེང་འདི་ན་སྐྱིད་སྡུག་སྣ་ཚོགས་མཐོང་། ས་སྟེང་དགེ་མཚན་གྱིས་ཁྱབ་ལྟ་བུ། ༼༢༽ སའི་ཐོག་ངོས་ཀྱི་མིང་སྟེ། ས་སྟེང་དུ་བསྡད། ས་སྟེང་བཞག་ལྟ་བུ། ༼ས་མཐའ༽ ༼༡༽ རང་གི་ཕ་ཡུལ་དང་རིང་བའི་ས་ཆ་སྟེ། སྔོན་ཆད་ས་མཐར་འཁྱམས། ས་མཐའི་མི་ཞེས་ལྟ་བུ། ༼༢༽ རྗེས་འཇུག་དང་ཡང་འཇུག་གི་ས་ཡིག་གི་མིང་། ༼ས་མདའ༽ བྱི་ལོང་དང་རི་དྭགས་སོགས་གསོད་བྱེད་ཀྱི་རྒྱ་རྙི་མདའ་གཞུ་ཅན་ཞིག་གི་མིང་། ༼ས་གདན༽ ཁང་བའི་ནང་གི་ས་ངོས་སུ་འདིང་རྒྱུའི་གདན། ༼ས་སྡེ་ལྔ༽ ཐོགས་མེད་ཀྱིས་མཛད་པའི་ས་སྡེ་ལྔ་སྟེ། སའི་དངོས་གཞི་དང་། གཏན་དབབ་པ་བསྡུ་བ། གཞི་བསྡུ་བ། རྣམ་གྲངས་བསྡུ་བ། སྒོ་བསྡུ་བ་སྟེ་ལྔའོ། ༼ས་འདུལ༽ གཙུག་ལག་ཁང་དང་ཁང་ཆེན་སོགས་བཞེངས་པའི་སྔོན་དུ་ས་གཞི་འདུལ་སྦྱོང་བྱེད་པའི་སྔགས་ཀྱི་ཆོ་ག་ཞིག ༼ས་འདོལ༽ ས་ངོས་ཀྱི་བྱེ་མའི་རྣམ་པ་ཅན་གྱི་ས། ༼ས་དཔྱད༽ ༡་ས་ཆའི་བཀོད་པའི་མཚན་རྟགས། ༢.ས་ཆའི་བཀོད་པ་ལེགས་ཉེས་བརྟག་དཔྱད་བྱེད་ཚུལ། ༣.ས་ཆའིབཀོད་པ་བརྟག་དཔྱད་ཀྱི་དཔེ་ཆ། ༼ས་སྦྱིན༽ སྤྱི་ན་བསྲེས་པའི་ས་འདམ་གྱི་མིང་། ༼ས་ཕག༽ ས་འདམ་གྱི་ལེབ་མ་གྲུ་བཞི་རིང་བོ་རྩིག་པའི་རྒྱུ་བྱེད་པ་ཞིག ༼ས་བབ༽ ༼ས་བླ༽ ༼༡༽ ས་ལས་འཕགས་པའམ་ས་དང་མ་འབྲེལ་བའི་བར་སྣང་། ས་གསུམ་གྱི་ཡ་གྱལ་ཞིག ༼༢༽ ལྷ་ཡུལ་ལ་ཡང་གོ ༼ས་བོགས༽ ས་ཞིང་དབང་བའི་དབང་ཆར་བརྟེན་ནས་ཐོབ་པའི་ཡོང་སྒོའི་མིང་སྟེ། ཐོང་ག་དང་དོན་འདྲ། ༼ས་ཚིགས༽ ནུས་པའམ་ཕོད་པ་སྟེ་རྒྱ་གར་གྱི་རིགས་རུས་ཤིག ༼ཤཱཀྱ་ཐུབ་པ༽ སངས་རྒྱས་ཤཱཀྱ་མུ་ནེ་སྟེ། ཆུང་དུས་མཚན་དོན་ཀུན་གྲུབ་པ་ཞེས་གྲགས། ཁོང་ནི་དྲི་མེད་ཤཱཀྱའི་རྒྱལ་རྒྱུད་ལས་ཡབ་ཟས་གཙང་མའི་རིགས་སུ་སྐུ་འཁྲུངས་པ་ཡིན་ལ། ས་མོ་ལུག་གི་ལོའི་ཆུ་སྟོད་ཟླ་བའི་ཉ་ལ་ཡུམ་སྒྱུ་འཕྲུལ་མའི་ལྷུམས་སུ་ཞུགས། ལྕགས་ཕོ་སྤྲེལ་ལོའི་ས་ག་ཟླ་བའི་ཚེས་བདུན་ལ་ལུམྦི་ནའི་ཚལ་དུ་སྐུ་བལྟམས། གཞོན་ནུའི་དུས་ནས་སྒྱུ་རྩལ་དྲུག་ཅུ་རྩ་བཞི་རང་དབང་དུ་བྱས། བུ་མོ་ས་འཚོ་མ་སོགས་ཁབ་ཏུ་བཞེས་ཏེ་བཙུན་མོའི་ཚོགས་ཀྱིས་བསྐོར་ཏེ་རྒྱལ་སྲིད་བསྐྱངས། དགུང་ལོ་ཉེར་དགུའི་ཐོག་ཁྱིམ་ནས་ཁྱིམ་མེད་པར་རབ་ཏུ་བྱུང་སྟེ་མཆོད་རྟེན་རྣམ་དག་གི་དྲུང་དུ་དཔལ་རང་བྱུང་གིས་བསྙེན་པར་རྫོགས་། ཆུ་བོ་ནཻ་རཉྫ་ནའི་འགྲམ་དུ་ལོ་དྲུག་ཏུ་དཀའ་བ་སྤྱད། དགུང་ལོ་སོ་ལྔ་པ་ཤིང་ཕོ་རྟའི་ལོའི་ས་ག་ཟླ་བའི་ཚེས་བཅོ་ལྔའི་སྲོད་ལ་བདུད་བཏུལ། གུང་ལ་མཉམ་པར་བཞག་ཐོ་རངས་ལ་མངོན་པར་རྫོགས་པར་སངས་རྒྱས། དེ་ནས་ཞག་བདུན་ཕྲག་བདུན་གྱི་མཐར་ཆུ་སྟོད་ཟླ་བའི་ཚེས་བཞིར་ཡུལ་ཝརཱ་ཎ་སིར་བཀའ་དང་པོ་བདེན་བཞིའི་ཆོས་འཁོར་བསྐོར། རིམ་བཞིན་རྒྱལ་པོའི་ཁབ་དང་། བྱ་རྒོད་ཕུང་པོ་ལ་སོགས་པར་བཀའ་བར་པ་མཚན་ཉིད་མེད་པ་དང་། བཀའ་ཐ་མ་ལེགས་པར་རྣམ་པར་ཕྱེ་བའོ་ཆོས་ཀྱི་འཁོར་ལོ་བསྐོར། མཐར་དགུང་ལོ་གྱ་གཅིག་པ་ལྕགས་འབྲུག་ས་ག་ཟླ་བའི་ཚེས་བཅོ་ལྔ་ལ་གནས་རྩྭ་མཆོག་དྲུང་དུ་སྐུ་མྱ་ངན་ལས་འདས་པའི་ཚུལ་བསྟན་པ་སོགས་མཛད་པ་ཆེན་པོ་བཅུ་གཉིས་ཡོད་དོ། ༼ཤཱཀྱའི་བསྟན་པ༽ སངས་རྒྱས་ཀྱི་ཆོས་ལུགས། ༼ཤཱཀྱའི་དགེ་སྦྱོང༽ སངས་རྒྱས་ཀྱི་ཆོས་ལུགས་པའི་དགེ་སློང་གིས་རང་ལ་དེ་སྐད་འབོད། ༼ཤཱཀྱ་འོད༽ རྒྱ་གར་གྱི་སློབ་དཔོན་རྒྱན་དྲུག་གི་ཡ་གྱལ་འདུལ་བ་བདུད་རྩིའི་རྒྱན་གྱི་སློབ་དཔོན་ཞིག་སྟེ། ཁོང་གིས་དགེ་ཚུལ་གྱི་ཀ་རི་ཀ་སུམ་བརྒྱ་བ་སོགས་མཛད།
